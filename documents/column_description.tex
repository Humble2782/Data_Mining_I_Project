\section{General}

The columns can be divided into two main groups:
\begin{enumerate}
    \item \textbf{Original Columns:} Data originating from the original French datasets and translated into English by \texttt{a\_rename.py}.
    \item \textbf{Engineered Features:} New columns calculated and added in \texttt{c\_feature\_engineering.py} to enrich the dataset.
\end{enumerate}

\section{Original Columns}
These columns describe the core information about the accident, location, vehicles, and users.

\subsection{General Accident Characteristics}
\begin{itemize}
    \item \textbf{\texttt{year}}: Year of the accident. \textit{(Original: \texttt{an})}
    \item \textbf{\texttt{hour\_minute}}: Exact hour and minute of the accident. \textit{(Original: \texttt{hrmn})}
    \item \textbf{\texttt{lighting\_condition}}: Lighting conditions under which the accident occurred. \textit{(Original: \texttt{lum})}
        \begin{itemize}
            \item \texttt{1}: Full day
            \item \texttt{2}: Twilight or dawn
            \item \texttt{3}: Night without public lighting
            \item \texttt{4}: Night with public lighting not on
            \item \texttt{5}: Night with public lighting on
        \end{itemize}
    \item \textbf{\texttt{location}}: Location of the accident. \textit{(Original: \texttt{agg})}
        \begin{itemize}
            \item \texttt{1}: Outside built-up area
            \item \texttt{2}: In built-up area
        \end{itemize}
    \item \textbf{\texttt{intersection}}: Type of intersection. \textit{(Original: \texttt{int})}
        \begin{itemize}
            \item \texttt{1}: Outside intersection
            \item \texttt{2}: X-intersection
            \item \texttt{3}: T-intersection
            \item \texttt{4}: Y-intersection
            \item \texttt{5}: Intersection with >4 branches
            \item \texttt{6}: Roundabout
            \item \texttt{7}: Place / Square
            \item \texttt{8}: Level crossing
            \item \texttt{9}: Other intersection
        \end{itemize}
    \item \textbf{\texttt{weather\_condition}}: Atmospheric conditions. \textit{(Original: \texttt{atm})}
        \begin{itemize}
            \item \texttt{-1}: Not specified
            \item \texttt{1}: Normal
            \item \texttt{2}: Light rain
            \item \texttt{3}: Heavy rain
            \item \texttt{4}: Snow - hail
            \item \texttt{5}: Fog - smoke
            \item \texttt{6}: Strong wind - storm
            \item \texttt{7}: Dazzling weather
            \item \texttt{8}: Overcast
            \item \texttt{9}: Other
        \end{itemize}
    \item \textbf{\texttt{type\_of\_collision}}: Type of collision. \textit{(Original: \texttt{col})}
        \begin{itemize}
            \item \texttt{-1}: Not specified
            \item \texttt{1}: Two vehicles - frontal
            \item \texttt{2}: Two vehicles - from behind
            \item \texttt{3}: Two vehicles - from the side
            \item \texttt{4}: Three vehicles and more - in chain
            \item \texttt{5}: Three vehicles and more - multiple collisions
            \item \texttt{6}: Other collision
            \item \texttt{7}: Without collision
        \end{itemize}
    \item \textbf{\texttt{latitude}}: Latitude (WGS84). \textit{(Original: \texttt{lat})}
    \item \textbf{\texttt{longitude}}: Longitude (WGS84). \textit{(Original: \texttt{long})}
\end{itemize}

\subsection{Location Characteristics}
\begin{itemize}

            \item \texttt{2}: Wet
            \item \texttt{3}: Puddles
            \item \texttt{4}: Flooded
            \item \texttt{5}: Snowy
            \item \texttt{6}: Mud
            \item \texttt{7}: Icy
            \item \texttt{8}: Grease - oil
            \item \texttt{9}: Other
        \end{itemize}
    \item \textbf{\texttt{infrastructure}}: Special infrastructure or facilities at the accident site. \textit{(Original: \texttt{infra})}
        \begin{itemize}
            \item \texttt{-1}: Not specified
            \item \texttt{0}: None
            \item \texttt{1}: Underground - tunnel
            \item \texttt{2}: Bridge - overpass
            \item \texttt{3}: Interchange ramp or connection
            \item \texttt{4}: Railway
            \item \texttt{5}: Equipped junction
            \item \texttt{6}: Pedestrian zone
            \item \texttt{7}: Toll zone
            \item \texttt{8}: Construction site
            \item \texttt{9}: Others
        \end{itemize}
    \item \textbf{\texttt{accident\_situation}}: Situation of the accident. \textit{(Original: \texttt{situ})}
        \begin{itemize}
            \item \texttt{-1}: Not specified
            \item \texttt{0}: None
            \item \texttt{1}: On roadway
            \item \texttt{2}: On emergency lane
            \item \texttt{3}: On shoulder
            \item \texttt{4}: On sidewalk
            \item \texttt{5}: On bicycle path
            \item \texttt{6}: On other special lane
            \item \texttt{8}: Others
        \end{itemize}
    \item \textbf{\texttt{speed\_limit}}: Authorized speed limit at the time and place of the accident. \textit{(Original: \texttt{vma})}
\end{itemize}

\subsection{User Characteristics (Occupants/Pedestrians)}
\begin{itemize}
    \item \textbf{\texttt{position}}: Position occupied by the user in the vehicle (e.g., \texttt{1}: Driver, \texttt{2}-\texttt{9}: Passenger seats). \textit{(Original: \texttt{place})}
    \item \textbf{\texttt{user\_category}}: Category of the user. \textit{(Original: \texttt{catu})}
        \begin{itemize}
            \item \texttt{1}: Driver
            \item \texttt{2}: Passenger
            \item \texttt{3}: Pedestrian
        \end{itemize}
     \item \textbf{\texttt{role}}: User's role, derived from \texttt{user\_category}. \textit{(Created in \texttt{b\_table\_merge.py})}
        \begin{itemize}
            \item \texttt{'driver'}: (From \texttt{user\_category} = 1)
            \item \texttt{'passenger'}: (From \texttt{user\_category} = 2)
            \item \texttt{'pedestrian'}: (From \texttt{user\_category} = 3)
            \item \texttt{'other'}: (For any other or null value)
        \end{itemize}
    \item \textbf{\texttt{injury\_severity}}: Severity of the user's injury. \textit{(Original: \texttt{grav})}
        \begin{itemize}
            \item \texttt{1}: Uninjured
            \item \texttt{2}: Killed
            \item \texttt{3}: Hospitalized injured
            \item \texttt{4}: Lightly injured
        \end{itemize}
    \item \textbf{\texttt{sex}}: Sex of the user. \textit{(Original: \texttt{sexe})}
        \begin{itemize}
            \item \texttt{1}: Male
            \item \texttt{2}: Female
        \end{itemize}
    \item \textbf{\texttt{pedestrian\_location}}: Location of the pedestrian at the time of the accident. \textit{(Original: \texttt{locp})}
        \begin{itemize}
            \item \texttt{-1}: Not specified
            \item \texttt{0}: Not applicable
            \item \texttt{1}: On roadway, >50m from pedestrian crossing
            \item \texttt{2}: On roadway, <50m from pedestrian crossing
            \item \texttt{3}: On pedestrian crossing, without traffic light
            \item \texttt{4}: On pedestrian crossing, with traffic light
            \item \texttt{5}: On sidewalk
            \item \texttt{6}: On shoulder
            \item \texttt{7}: On refuge or emergency lane
            \item \texttt{8}: On parallel lane
            \item \texttt{9}: Unknown
        \end{itemize}
    \item \textbf{\texttt{pedestrian\_action}}: Action of the pedestrian. \textit{(Original: \texttt{actp})}
        \begin{itemize}
            \item \texttt{-1}: Not specified
            \item \texttt{0}: Not specified or not applicable
            \item \texttt{1}: Moving in the same direction as the striking vehicle
            \item \texttt{2}: Moving in the opposite direction of the vehicle
            \item \texttt{3}: Crossing
            \item \texttt{4}: Masked / Hidden
            \item \texttt{5}: Playing - running
            \item \texttt{6}: With animal
            \item \texttt{9}: Other
            \item \texttt{A}: Getting on/off vehicle
            \item \texttt{B}: Unknown
        \end{itemize}
\end{itemize}

\section{Engineered Features (From c\_feature\_engineering.py)}
These columns were newly created to model more complex relationships.
\begin{itemize}
    \item \textbf{\texttt{vehicle\_category\_simplified}}: A simplified grouping of the detailed \texttt{vehicle\_category}. Possible values: \texttt{bicycle}, \texttt{powered\_2\_3\_wheeler}, \texttt{light\_motor\_vehicle}, \texttt{hgv\_truck}, \texttt{bus\_coach}, \texttt{other}, \texttt{unknown}.
    \item \textbf{\texttt{impact\_score}}: A calculated "impact score" (weight) based on the \texttt{vehicle\_category\_simplified} (e.g., \texttt{hgv\_truck}: 6, \texttt{light\_motor\_vehicle}: 4, \texttt{bicycle}: 2).
    \item \textbf{\texttt{age}}: The user's age at the time of the accident, calculated as \texttt{year} - \texttt{year\_of\_birth}.
    \item \textbf{\texttt{age\_group}}: An age bracket based on the \texttt{age}: \texttt{child\_teen} (0-17), \texttt{young\_adult} (18-24), \texttt{adult} (25-39), \texttt{middle\_aged} (40-64), \texttt{senior} (65+).
    \item \textbf{\texttt{used\_belt}}: A binary indicator (1 or 0) specifying if the user used a seatbelt (based on \texttt{safety\_equipment\_1}, \texttt{\_2}, \texttt{\_3} columns, where the value \texttt{1} means "Belt").
    \item \textbf{\texttt{used\_helmet}}: A binary indicator (1 or 0) specifying if the user used a helmet (based on \texttt{safety\_equipment\_1}, \texttt{\_2}, \texttt{\_3} columns, where the value \texttt{2} means "Helmet").
    \item \textbf{\texttt{day\_of\_week\_sin}} / \textbf{\texttt{day\_of\_week\_cos}}: Cyclical features (Sine/Cosine) representing the day of the week to model weekly seasonality.
    \item \textbf{\texttt{month\_sin}} / \textbf{\texttt{month\_cos}}: Cyclical features (Sine/Cosine) representing the month to model annual seasonality.
    \item \textbf{\texttt{day\_of\_year\_sin}} / \textbf{\texttt{day\_of\_year\_cos}}: Cyclical features (Sine/Cosine) representing the day of the year (more granular annual seasonality).
    \item \textbf{\texttt{road\_complexity\_index}}: An index (scaled from 0-10) that assesses the complexity of the road. It is based on a weighted sum of \texttt{intersection}, \texttt{road\_category}, \texttt{traffic\_regime}, and \texttt{number\_of\_traffic\_lanes}.
    \item \textbf{\texttt{surface\_quality\_indicator}}: A binary indicator (\texttt{1}=good, \texttt{0}=bad). It is only \texttt{1} (good) if the \texttt{pavement\_condition} is "Normal" (1) AND the \texttt{longitudinal\_profile} is "Flat" (1).
\end{itemize}
